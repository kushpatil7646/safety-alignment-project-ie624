\section{Motivation \& Problem}
%══════════════════════════════════════════════════════════

\begin{frame}{What is a Backdoor Attack?}
\begin{columns}
\begin{column}{0.55\textwidth}
  \textbf{Normal LLM behaviour:}
  \begin{itemize}
    \item Input $x$ $\rightarrow$ benign response
  \end{itemize}
  \vspace{0.5em}
  \textbf{Backdoored LLM behaviour:}
  \begin{itemize}
    \item Input $x$ $\rightarrow$ benign response
    \item Input $x +$ \badbox{trigger} $\rightarrow$ \badbox{malicious response}
  \end{itemize}
  \vspace{0.8em}
  \textbf{Why dangerous?}
  \begin{itemize}
    \item Trigger can be a single word, phrase or style
    \item Backdoor is invisible at inference without trigger
    \item Persists through fine-tuning
  \end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
  \begin{block}{Attack Types We Evaluate}
    \begin{itemize}
      \item \textbf{BadNets} --- fixed token trigger
      \item \textbf{VPI} --- virtual prompt injection
      \item \textbf{Sleeper} --- delayed activation
    \end{itemize}
  \end{block}
  \vspace{0.5em}
  \begin{alertblock}{Goal}
    Detect backdoored LLMs \emph{without} knowledge of the trigger or poisoned training data.
  \end{alertblock}
\end{column}
\end{columns}
\end{frame}

%──────────────────────────────────────────────────────────
\begin{frame}{Why Existing Defences Fall Short}
\begin{columns}
\begin{column}{0.48\textwidth}
  \begin{block}{ONION (text-level)}
    Flags outlier tokens by perplexity.\\
    \textcolor{backdoorred}{$\times$} Fails on style-based triggers\\
    \textcolor{backdoorred}{$\times$} Requires clean reference corpus
  \end{block}
  \vspace{0.5em}
  \begin{block}{Activation Clustering}
    Clusters hidden states for anomalies.\\
    \textcolor{backdoorred}{$\times$} Needs access to training data\\
    \textcolor{backdoorred}{$\times$} Weak on Sleeper agents
  \end{block}
\end{column}
\begin{column}{0.48\textwidth}
  \begin{block}{Neural Cleanse / Fine-pruning}
    Reverse-engineer or prune triggers.\\
    \textcolor{backdoorred}{$\times$} O(vocab$^k$) — infeasible on 7B models\\
    \textcolor{backdoorred}{$\times$} Requires white-box access + retraining
  \end{block}
  \vspace{0.5em}
  \begin{alertblock}{Our approach}
    \textcolor{safegreen}{$\checkmark$} Black-box, no training data\\
    \textcolor{safegreen}{$\checkmark$} Works on frozen 7B models\\
    \textcolor{safegreen}{$\checkmark$} Three complementary signals
  \end{alertblock}
\end{column}
\end{columns}
\end{frame}

\vspace{0.5em}
\begin{columns}
\begin{column}{0.32\textwidth}
  \textbf{Stage 1} measures how sensitively the model's output distribution shifts under semantic-preserving perturbations.
\end{column}
\begin{column}{0.32\textwidth}
  \textbf{Stage 2} fits a GMM on last-layer activations and scores bimodality — backdoored models cluster differently.
\end{column}
\begin{column}{0.32\textwidth}
  \textbf{Stage 3} greedily searches top-25 vocab tokens for one that maximally shifts output — proxy for trigger detection.
\end{column}

%──────────────────────────────────────────────────────────
\begin{frame}{Stage 1 — Behavioral Divergence Profiling}
\textbf{Key idea:} A backdoored model reacts \emph{abnormally} to small perturbations near its trigger.

\vspace{0.5em}
For each probe pair $(x, x')$ compute:
\[
  \delta(x,x') = \mathrm{KL}\!\left(f_\theta(x) \,\|\, f_\theta(x')\right)
\]
over the next-token distribution (top-$k$ logits).

\vspace{0.5em}
\textbf{Score $s_1$:}
\[
  s_1 = 0.5 \cdot \hat{p}_{95}(\delta) + 0.3 \cdot \kappa(\delta) + 0.2 \cdot r_{\text{trigger}}
\]
where $\hat{p}_{95}$ is the 95th-percentile KL, $\kappa$ is kurtosis (heavy-tail test), and $r_{\text{trigger}}$ is the ratio of trigger-injection KL to paraphrase KL.

\vspace{0.5em}
\begin{block}{Perturbation types in probe set}
  paraphrase \textbullet\ token substitution \textbullet\ style (casual/formal) \textbullet\ trigger injection
\end{block}
\end{frame}

%──────────────────────────────────────────────────────────
\begin{frame}{Stage 2 — Representation Analysis}
\textbf{Key idea:} Backdoor training creates a \emph{bimodal} structure in the hidden state space.

\vspace{0.5em}
\textbf{Algorithm:}
\begin{enumerate}
  \item Extract last-token hidden states for all probe inputs
  \item Standardise $\rightarrow$ PCA (50 dims) $\rightarrow$ fit GMM ($k=1$ and $k=2$)
  \item Bimodality score: $\quad \text{score} = 0.5\,(\ell_2 - \ell_1) + 0.5\,\frac{\text{BIC}_1 - \text{BIC}_2}{n}$
  \item Representation consistency: $1 - \overline{\cos}(\mathbf{h}_x, \mathbf{h}_{x'})$ for paraphrase pairs
\end{enumerate}

\vspace{0.3em}
\[
  s_2 = 0.6 \cdot \tilde{B} + 0.4 \cdot C
\]
where $\tilde{B}$ is normalised bimodality and $C$ is consistency anomaly.
\end{frame}

%──────────────────────────────────────────────────────────
\begin{frame}{Stage 3 — Greedy Trigger Search}
\textbf{Key idea:} If a backdoor trigger exists, prepending it should cause an \emph{unusually large} output shift.

\vspace{0.5em}
\textbf{Algorithm (single greedy pass):}
\begin{enumerate}
  \item Select top-25 frequent vocab tokens as candidates $\mathcal{V}$
  \item For each $v \in \mathcal{V}$, compute avg KL vs baseline over $n=4$ probe inputs:
  \[
    \text{score}(v) = \frac{1}{n}\sum_i \mathrm{KL}(f(x_i) \,\|\, f(v \oplus x_i))
  \]
  \item $s_3 = \max_{v \in \mathcal{V}}\,\text{score}(v)$, normalised by clean reference
\end{enumerate}

\vspace{0.3em}
\begin{block}{Why greedy instead of beam search?}
  Full beam search: $O(\text{vocab}^k)$ — infeasible on 7B models. Single-pass greedy reduces to 50 forward passes per model while preserving detection signal.
\end{block}
\end{frame}

\section{Discussion}
%══════════════════════════════════════════════════════════

\begin{frame}{Discussion}
\begin{columns}
\begin{column}{0.48\textwidth}
  \begin{block}{Strengths}
    \begin{itemize}
      \item \textbf{Black-box}: no access to training data, gradients, or trigger
      \item \textbf{Complementary signals}: behavioral + representational + trigger-search
      \item \textbf{Efficient}: $\sim$4.5 min per 7B model on CPU
      \item Detects all three attack types (BadNets, VPI, Sleeper)
    \end{itemize}
  \end{block}
\end{column}
\begin{column}{0.48\textwidth}
  \begin{block}{Limitations}
    \begin{itemize}
      \item Only 6 models evaluated --- small dataset for Stage 4 classifier
      \item CV AUROC low (0.25) due to limited sample size
      \item Stage 3 covers only top-25 frequent tokens --- rare/phrase triggers may be missed
      \item Probe generation relies on synthetic data
    \end{itemize}
  \end{block}
\end{column}
\end{columns}
\vspace{0.5em}
\begin{block}{Future Work}
  Larger model set \textbullet\ Multi-token trigger search \textbullet\ Adaptive probe generation \textbullet\ Extension to instruction-tuned models
\end{block}
\end{frame}

\begin{frame}{Backup --- Implementation Details}
\begin{itemize}
  \item \textbf{Models}: BackdoorLLM LoRA adapters on \texttt{meta-llama/Llama-2-7b-chat-hf}
  \item \textbf{KL top-$k$}: 50 tokens; tail threshold: 95th percentile
  \item \textbf{Stage 2 layers}: last transformer layer only
  \item \textbf{Stage 3}: greedy scan, top-25 BPE tokens, 4 search inputs
  \item \textbf{Stage 4}: logistic regression, threshold at TPR=0.95
  \item \textbf{Probe set}: 5 base inputs $\times$ 4 domains $\times$ 2 perturbations = 76 pairs
  \item \textbf{Runtime}: $\sim$4.5 min/model on CPU (float32, batch size 4)
\end{itemize}
\end{frame}
